\Def{滤子}
设 $X$ 是集合 $S$ 的一个子集, 且满足以下条件:
\begin{itemize}
\item $X$ 非空;
\item 若 $A\in X$ 且 $A\subseteq B\subseteq S$, 则 $B\in X$;
\item 若 $A\in X$ 且 $B\in X$, 则 $A\cap B\in X$.
\end{itemize}
则称 $X$ 是集合 $S$ 上的一个滤子。
\Thm{Stolz 定理}
设 $\{a_n\}$ 和 $\{b_n\}$ 是两个数列, 且 $\{b_n\}$ 是一个严格单调递增且无界的数列. 若 $\lim_{n\to\infty} \frac{a_{n+1}-a_n}{b_{n+1}-b_n} = L$ 存在, 则 $\lim_{n\to\infty} \frac{a_n}{b_n} = L$.

\begin{algorithm}
\caption{recursive Fibonacci}
\label{alg:fibonacci}
\textbf{Input:} $n \in \mathrm{N}$\\
\textbf{Output:} $F(n)$
\begin{enumerate}
\item If $n=0$, return $0$.
\item If $n=1$, return $1$.
\item Otherwise, return $\textit{Fibonacci}(n-1)+\textit{Fibonacci}(n-2)$.
\end{enumerate}
\end{algorithm}
